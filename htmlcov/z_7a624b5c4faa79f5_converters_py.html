25def harbor_to_compose_config( 

26 harbor_task: HarborTask, 

27 override_cpus: int | None = None, 

28 override_memory_mb: int | None = None, 

29 override_gpus: int | None = None, 

30) -> ComposeConfig: 

31 """Convert Harbor task environment to Inspect ComposeConfig. 

32 

33 Args: 

34 harbor_task: The Harbor task to convert. 

35 override_cpus: Override the number of CPUs for the default service. 

36 override_memory_mb: Override the memory (in MB) for the default service. 

37 override_gpus: Override the number of GPUs for the default service. 

38 

39 Returns: 

40 ComposeConfig: The compose configuration for the task. 

41 """ 

42 env_dir = harbor_task.paths.environment_dir 

43 compose_yaml_path = env_dir / "docker-compose.yaml" 

44 dockerfile_path = env_dir / "Dockerfile" 

45 env_config = harbor_task.config.environment 

46 

47 # Extract resource configuration from Harbor config, applying overrides 

48 cpus = float(override_cpus) if override_cpus is not None else float(env_config.cpus) 

49 

50 # Harbor's default of 2048 MB (2 GB) is too restrictive for modern agents. 

51 # Enforce a minimum of 6144 MB (6 GB) unless explicitly overridden. 

52 MIN_MEMORY_MB = 6144 # 6 GB 

53 memory_mb = ( 

54 override_memory_mb 

55 if override_memory_mb is not None 

56 else max(env_config.memory_mb, MIN_MEMORY_MB) 

57 ) 

58 

59 gpus = override_gpus if override_gpus is not None else env_config.gpus 

60 gpu_deploy = _create_gpu_deploy_config(gpus, env_config.gpu_types) 

61 

62 # Use existing docker-compose.yaml if present 

63 if compose_yaml_path.exists(): 

64 with open(compose_yaml_path, encoding="utf-8") as f: 

65 raw_yaml = f.read() 

66 

67 raw_yaml = _expand_compose_vars(raw_yaml, harbor_task, cpus, memory_mb) 

68 compose_dict = yaml.safe_load(raw_yaml) 

69 compose_config = ComposeConfig(**compose_dict)

72 _, default_service = _find_default_service(compose_config)

75 if gpu_deploy: 

76 default_service.deploy = gpu_deploy 

77 

78 # Network isolation applies to all services. 

79 if not env_config.allow_internet: 

80 for service in compose_config.services.values(): 

81 service.network_mode = "none"

107def harbor_task_to_sample( 

108 harbor_task: HarborTask, 

109 sandbox_env_name: str = "docker", 

110 override_cpus: int | None = None, 

111 override_memory_mb: int | None = None, 

112 override_gpus: int | None = None, 

113 sample_id: str | None = None, 

114) -> Sample: 

115 """Convert a Harbor task to an Inspect AI Sample. 

116 

117 Args: 

118 harbor_task: The Harbor task to convert. 

119 sandbox_env_name: Sandbox environment name (default: ``docker``). 

120 override_cpus: Override the number of CPUs for the environment. 

121 override_memory_mb: Override the memory (in MB) for the environment. 

122 override_gpus: Override the number of GPUs for the environment. 

123 sample_id: Override the resulting ``Sample.id``. Defaults to 

124 ``harbor_task.name`` when ``None``. Used by the loader to 

125 disambiguate samples whose Harbor names collide. 

126 

127 Returns: 

128 Sample: Inspect AI sample with sandbox configuration. 

129 """ 

130 compose_config = harbor_to_compose_config( 

131 harbor_task, 

132 override_cpus=override_cpus, 

133 override_memory_mb=override_memory_mb, 

134 override_gpus=override_gpus, 

135 ) 

136 

137 metadata: dict[str, Any] = { 

138 "task_name": harbor_task.name, 

139 "task_dir": str(harbor_task.task_dir), 

140 "test_path": str(harbor_task.paths.test_path), 

141 "tests_dir": str(harbor_task.paths.tests_dir), 

142 "solution_dir": str(harbor_task.paths.solution_dir), 

143 "solve_path": str(harbor_task.paths.solve_path), 

144 "verifier_timeout_sec": harbor_task.config.verifier.timeout_sec, 

145 "verifier_env": harbor_task.config.verifier.env, 

146 "solution_env": harbor_task.config.solution.env, 

147 "verifier_user": _user_to_str(harbor_task.config.verifier.user), 

148 "agent_user": _user_to_str(harbor_task.config.agent.user), 

149 "harbor_config": harbor_task.config.model_dump(), 

150 } 

151 

152 if harbor_task.config.task is not None: 

153 package_info = harbor_task.config.task 

154 metadata["package_name"] = package_info.name 

155 metadata["package_description"] = package_info.description 

156 metadata["package_keywords"] = list(package_info.keywords) 

157 metadata["package_authors"] = [a.model_dump() for a in package_info.authors] 

158 

159 return Sample( 

160 input=harbor_task.instruction, 

161 id=sample_id if sample_id is not None else harbor_task.name, 

162 sandbox=SandboxEnvironmentSpec(sandbox_env_name, compose_config), 

163 metadata=metadata, 

164 )

172def _find_default_service(config: ComposeConfig) -> tuple[str, ComposeService]: 

173 """Find the default service in a compose config. 

174 

175 Priority: ``x-default: true`` > service named "default" or "main" > first. 

176 """ 

177 for name, svc in config.services.items(): 

178 if svc.x_default: 

179 return name, svc 

180 for candidate in ("default", "main"): 

181 if candidate in config.services: 

182 return candidate, config.services[candidate] 

183 name = next(iter(config.services)) 

184 return name, config.services[name]

187def _create_gpu_deploy_config( 

188 gpus: int, gpu_types: list[str] | None 

189) -> ComposeDeploy | None: 

190 """Create GPU deployment configuration for ComposeService. 

191 

192 Args: 

193 gpus: Number of GPUs to reserve (0 means no GPUs). 

194 gpu_types: List of acceptable GPU types (e.g., ['H100', 'A100']). 

195 Stored in device options for informational purposes. 

196 

197 Returns: 

198 ComposeDeploy configuration with GPU reservations, or None if gpus=0. 

199 """ 

200 if gpus <= 0: 

201 return None 

202 

203 device_options = {} 

204 if gpu_types: 

205 # Store GPU types in options for potential use by sandbox providers 

206 device_options["gpu_types"] = ",".join(gpu_types) 

207 

208 device_reservation = ComposeDeviceReservation( 

209 count=gpus, 

210 capabilities=["gpu"], 

211 options=device_options if device_options else None, 

212 ) 

213 

214 return ComposeDeploy( 

215 resources=ComposeResourceConfig( 

216 reservations=ComposeResourceReservations(devices=[device_reservation]) 

217 ) 

218 )

221def _expand_compose_vars( 

222 raw_yaml: str, 

223 harbor_task: HarborTask, 

224 cpus: float, 

225 memory_mb: int, 

226) -> str: 

227 """Expand ``${VAR}`` and ``${VAR:-default}`` references in a Harbor docker-compose.yaml. 

228 

229 Harbor passes these variables as environment variables to the 

230 ``docker compose`` process, which performs the substitution natively. 

231 

232 Limitation: ``HOST_*`` paths (the host side of volume mounts) are set to 

233 the same container-side ``EnvironmentPaths`` values as ``ENV_*``. In 

234 Harbor's DinD setup these differ, but we cannot resolve host-side paths 

235 here because they depend on the sandbox provider. 

236 """ 

237 if "${" not in raw_yaml: 

238 return raw_yaml 

239 

240 env_dir = str(harbor_task.paths.environment_dir) 

241 paths = EnvironmentPaths() 

242 

243 var_map: dict[str, str] = { 

244 "CONTEXT_DIR": env_dir, 

245 # Mirror Harbor's own ``hb__<task.name>`` + ``_sanitize_docker_image_name`` 

246 # so package tasks (org/name) produce the same image name Harbor would. 

247 "MAIN_IMAGE_NAME": _sanitize_docker_image_name(f"hb__{harbor_task.name}"), 

248 "CPUS": str(int(cpus)), 

249 "MEMORY": f"{memory_mb}M", 

250 "HOST_VERIFIER_LOGS_PATH": str(paths.verifier_dir), 

251 "HOST_AGENT_LOGS_PATH": str(paths.agent_dir), 

252 "HOST_ARTIFACTS_PATH": str(paths.artifacts_dir), 

253 "ENV_VERIFIER_LOGS_PATH": str(paths.verifier_dir), 

254 "ENV_AGENT_LOGS_PATH": str(paths.agent_dir), 

255 "ENV_ARTIFACTS_PATH": str(paths.artifacts_dir), 

256 } 

257 

258 verifier_env = harbor_task.config.verifier.env 

259 if "TEST_DIR" in verifier_env: 

260 var_map["TEST_DIR"] = verifier_env["TEST_DIR"] 

261 else: 

262 var_map["TEST_DIR"] = str(paths.tests_dir) 

263 

264 task_env = harbor_task.config.environment.env 

265 for key, value in task_env.items(): 

266 var_map.setdefault(key, value)

268 def _replace(match: re.Match[str]) -> str: 

269 body = match.group(1) 

270 if ":-" in body: 

271 var_name, default = body.split(":-", 1) 

272 else: 

273 var_name, default = body, None 

274 if var_name in var_map: 

275 return var_map[var_name] 

276 if default is not None: 

277 return default 

278 return match.group(0) 

279 

280 return re.sub(r"\$\{([^}]+)}", _replace, raw_yaml)
